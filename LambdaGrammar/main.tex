\section{Grammar}
\begin{grammar}
    <term> ::= '\textbackslash' bindings '.' term
    \alt app

    <app> ::= atom
    \alt app atom

    <atom> ::= var
    \alt '(' term ')'

    <var> ::= NAME

    <bindings> ::= NAME+

    <NAME> ::= ['a'..'z']
\end{grammar}
